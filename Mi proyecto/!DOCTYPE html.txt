        sedesArray.forEach(s => ramasArray.forEach(r => tiposArray.forEach(t => supArray.forEach(sup => {
            let benRandom = [];
            if(Math.random() > 0.5) benRandom.push('techo');
            if(Math.random() > 0.3) benRandom.push('parqueo');
            if(Math.random() > 0.4) benRandom.push('gradas');
            let pRandom = t === 'alquiler' ? 40 : (t === 'reto' ? 25 : 8);
            pRandom += Math.floor(Math.random() * 20);

            let imgRandom = sup === 'arena' ? "https://images.unsplash.com/photo-1510566337590-2fc1f21d0faa?q=80&w=600" : (sup === 'natural' ? "https://images.unsplash.com/photo-1529900748604-07564a03e7a6?q=80&w=600" : "https://images.unsplash.com/photo-1574629810360-7efbbe195018?q=80&w=600");

            canchasBase.push({
                id: idExt++, n: `Cancha ${r.toUpperCase()} ${sup} en ${s}`, s: s, r: r, t: t, sup: sup, ben: benRandom, p: pRandom, rate: (3.8 + Math.random()).toFixed(1), rev: Math.floor(Math.random()*100), img: imgRandom, urg: "", urg_t: ""
            });
        }))));

        let canchasActuales = [...canchasBase];
        let cart = [];

        function renderCanchas(data) {
            const grid = document.getElementById('main-grid');
            document.getElementById('result-title').innerText = `Encontramos ${data.length} opciones`;
            
            if(data.length === 0) {
                grid.innerHTML = `<div style="padding: 40px; text-align:center; color:#888; font-size:1.2rem;">No hay canchas que coincidan con estos filtros. Intenta cambiar tu búsqueda.</div>`;
                return;
            }

            // Renderizamos solo las primeras 30 para no saturar la web, pero asegurando que la base responda al filtro.
            const aMostrar = data.slice(0, 30);

            grid.innerHTML = aMostrar.map(c => `
                <div class="card">
                    <div class="card-img-wrapper">
                        ${c.urg ? `<span class="badge-${c.urg_t}">${c.urg}</span>` : ''}
                        <img src="${c.img}" class="card-img">
                    </div>
                    <div class="card-body">
                        <div class="card-header">
                            <div class="card-title">
                                <h3>${c.n}</h3>
                                <p><i class="fas fa-map-marker-alt" style="color:var(--accent)"></i> Sede ${c.s.toUpperCase()}</p>
                            </div>
                            <div class="rating">
                                <div style="display:flex; flex-direction:column;">
                                    <span style="font-weight:900; color:#fff;">Excelente</span>
                                    <span class="rating-text">${c.rev} opiniones</span>
                                </div>
                                <div class="rating-score">${c.rate}</div>
                            </div>
                        </div>
                        
                        <div class="features">
                            <span class="feature-tag"><i class="fas fa-futbol"></i> ${c.r === 'fem' ? 'Femenino' : c.r === 'mix' ? 'Mixto' : 'Masculino'}</span>
                            <span class="feature-tag">${c.t.toUpperCase()}</span>
                            <span class="feature-tag">${c.sup.toUpperCase()}</span>
                            ${c.ben.includes('techo') ? `<span class="feature-tag"><i class="fas fa-umbrella"></i> Techada</span>` : ''}
                            ${c.ben.includes('parqueo') ? `<span class="feature-tag"><i class="fas fa-car"></i> Parqueo</span>` : ''}
                        </div>
                        
                        <div class="free-cancel"><i class="fas fa-check-circle"></i> Cancelación gratis antes de 24h</div>

                        <div class="card-footer">
                            <div class="price-box">
                                <span class="price-sub">Precio base</span><br>
                                <span class="price">$${c.p}.00</span>
                            </div>
                            <button class="btn btn-add ${c.r === 'fem' ? 'fem' : ''}" onclick="addToCart(${c.id}, '${c.n}', ${c.p}, '${c.r}')">Reservar</button>
                        </div>
                    </div>
                </div>
            `).join('');
        }

        function filtrar() {
            const s = document.getElementById('s-sede').value;
            const r = document.getElementById('s-rama').value;
            const t = document.getElementById('s-tipo').value;

            const cbSup = Array.from(document.querySelectorAll('.cb-sup:checked')).map(cb => cb.value);
            const cbBen = Array.from(document.querySelectorAll('.cb-ben:checked')).map(cb => cb.value);

            canchasActuales = canchasBase.filter(c => {
                const matchBuscador = (s === 'all' || c.s === s) && (r === 'all' || c.r === r) && (t === 'all' || c.t === t);
                const matchSup = cbSup.length === 0 || cbSup.includes(c.sup);
                const matchBen = cbBen.length === 0 || cbBen.every(b => c.ben.includes(b));
                return matchBuscador && matchSup && matchBen;
            });

            ordenar();
        }

        function ordenar() {
            const sortVal = document.getElementById('sort-select').value;
            if(sortVal === 'p-asc') canchasActuales.sort((a,b) => a.p - b.p);
            else if(sortVal === 'p-desc') canchasActuales.sort((a,b) => b.p - a.p);
            else if(sortVal === 'rating') canchasActuales.sort((a,b) => b.rate - a.rate);
            else canchasActuales.sort((a,b) => a.id - b.id);
            renderCanchas(canchasActuales);
        }

        function addToCart(id, nombre, precio, tipo) {
            cart.push({ cartId: Date.now(), id, nombre, precio, tipo });
            updateCart();
            showToast(`Añadido al carrito`);
        }

        function removeCartItem(cartId) {
            cart = cart.filter(i => i.cartId !== cartId);
            updateCart();
        }

        function updateCart() {
            document.getElementById('cart-count').innerText = cart.length;
            const list = document.getElementById('cart-list');
            
            list.innerHTML = cart.map(i => `
                <div class="cart-item ${i.tipo === 'fem' ? 'fem' : ''}">
                    <div>
                        <h4 style="font-size:1.1rem; margin-bottom:4px;">${i.nombre}</h4>
                        <p style="font-size:0.8rem; color:#888;">${i.tipo === 'store' ? 'Equipamiento' : 'Cancha'}</p>
                    </div>
                    <div style="text-align:right;">
                        <span style="font-weight:900; font-size:1.1rem; color:var(--accent); display:block;">$${i.precio}.00</span>
                        <i class="fas fa-trash" style="color:#ff4d4d; cursor:pointer; margin-top:8px;" onclick="removeCartItem(${i.cartId})"></i>
                    </div>
                </div>
            `).join('');

            const total = cart.reduce((sum, item) => sum + item.precio, 0);
            document.getElementById('cart-total').innerText = `$${total}.00`;
        }

        function toggleCart() { 
            document.getElementById('cart-panel').classList.toggle('open'); 
            const overlay = document.getElementById('cart-overlay');
            overlay.style.display = overlay.style.display === 'block' ? 'none' : 'block';
        }

        // NUEVO: Función integrada de WhatsApp + Excel
        function procesarReservaTotal() {
            const nombre = document.getElementById('cliente-nombre').value.trim();
            const telefono = document.getElementById('cliente-telefono').value.trim();
            const fecha = document.getElementById('s-fecha').value || "Pendiente";
            
            if(!nombre || !telefono) return alert("Por favor, ingresa tu Nombre y WhatsApp antes de reservar.");
            if(cart.length === 0) return alert("El carrito está vacío.");

            const total = cart.reduce((sum, item) => sum + item.precio, 0);

            // 1. Guardar y Descargar en Excel para el Administrador
            guardarEnExcel(nombre, telefono, fecha, total);

            // 2. Construir el mensaje y abrir WhatsApp
            let msg = `*RESERVA MATCHPOINT SMR*%0A---------------------------%0A*A nombre de:* ${nombre}%0A*Teléfono:* ${telefono}%0A*Fecha solicitada:* ${fecha}%0A---------------------------%0A*DETALLE DEL PEDIDO:*%0A`;
            
            cart.forEach(i => msg += `• ${i.nombre} -> $${i.precio}.00%0A`);
            
            msg += `---------------------------%0A*TOTAL: $${total}.00*%0A%0A_Por favor confirmar disponibilidad._`;
            
            // Retraso para dejar que el Excel se descargue antes de cambiar de app/pestaña
            setTimeout(() => {
                window.open(`https://wa.me/573159715828?text=${msg}`, '_blank');
            }, 800);
        }

        // NUEVO: Generador de Excel (.csv)
        function guardarEnExcel(nombre, telefono, fecha, total) {
            let csvContent = "Fecha_Registro,Nombre_Cliente,Telefono,Fecha_Reserva,Detalle_Pedido,Total_SMR\n";
            const fechaHoy = new Date().toLocaleString('es-CO');
            const nombresItems = cart.map(i => i.nombre).join(" + ");
            
            // Comillas para evitar problemas con las comas dentro del texto
            const fila = `"${fechaHoy}","${nombre}","${telefono}","${fecha}","${nombresItems}","${total}"\n`;
            csvContent += fila;

            const blob = new Blob([new Uint8Array([0xEF, 0xBB, 0xBF]), csvContent], { type: 'text/csv;charset=utf-8;' });
            const url = URL.createObjectURL(blob);
            const link = document.createElement("a");
            const fileName = `Reserva_MatchPoint_${nombre.replace(/\s+/g, '_')}.csv`;

            link.setAttribute("href", url);
            link.setAttribute("download", fileName);
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }

        function showToast(msg) {
            const t = document.getElementById('toast');
            t.innerText = msg; 
            t.classList.add('show');
            setTimeout(() => t.classList.remove('show'), 2500);
        }
